
======================
USERS MODULE
======================
page -- 
# Get user profile

  -H "Authorization: Bearer <ACCESS_TOKEN>"

page --

  -H "Authorization: Bearer <ACCESS_TOKEN>"

  -H "Authorization: Bearer <ACCESS_TOKEN>"

======================
QUESTIONS MODULE
======================

page -- 
# Get all questions
curl -X GET "http://localhost:5000/api/questions" \
  -H "Authorization: Bearer <ACCESS_TOKEN>"


page -- 5
# Get question by ID
curl -X GET "http://localhost:5000/api/questions/<QUESTION_ID>" \
  -H "Authorization: Bearer <ACCESS_TOKEN>"


page -- 3
# Create question (teacher/admin)
curl -X POST "http://localhost:5000/api/questions" \
  -H "Authorization: Bearer <TEACHER_TOKEN>" \
  -H "Content-Type: application/json" \
  -d '{"question":"What is 2+2?","q_type":"mcq","options":["1","2","3","4"],"correct_answer":"4","max_marks":1,"subject":"Math"}'


page -- 3
# Update question
curl -X PATCH "http://localhost:5000/api/questions/<QUESTION_ID>" \
  -H "Authorization: Bearer <TEACHER_TOKEN>" \
  -H "Content-Type: application/json" \
  -d '{"remarks":"Updated wording"}'


page -- 3
# Soft-delete question
curl -X DELETE "http://localhost:5000/api/questions/<QUESTION_ID>" \
  -H "Authorization: Bearer <TEACHER_TOKEN>"

MULTIPLAYER MODULE
======================

page -- 9
# Create lobby
curl -X POST "http://localhost:5000/api/multiplayer/create" \
  -H "Authorization: Bearer <TEACHER_ACCESS_TOKEN>" \
  -H "Content-Type: application/json" \
  -d '{"quiz_id":"<QUIZ_ID>"}'

page -- 10
# Join lobby
curl -X POST "http://localhost:5000/api/multiplayer/join" \
  -H "Authorization: Bearer <STUDENT_ACCESS_TOKEN>" \
  -H "Content-Type: application/json" \
  -d '{"join_code":"<JOIN_CODE>"}'

page -- 10
# Get lobby info
curl -X GET "http://localhost:5000/api/multiplayer/<SESSION_ID>" \
  -H "Authorization: Bearer <ACCESS_TOKEN>"


page -- 9
# Start session (host)
curl -X POST "http://localhost:5000/api/multiplayer/<SESSION_ID>/start" \
  -H "Authorization: Bearer <TEACHER_ACCESS_TOKEN>"

page -- 
# Submit multiplayer answers
curl -X POST "http://localhost:5000/api/multiplayer/<SESSION_ID>/submit" \
  -H "Authorization: Bearer <STUDENT_ACCESS_TOKEN>" \
  -H "Content-Type: application/json" \
  -d '{"answers":{"q_1":"A","q_2":"B"}}'


page -- 11
# End session (host)
curl -X POST "http://localhost:5000/api/multiplayer/<SESSION_ID>/end" \
  -H "Authorization: Bearer <TEACHER_ACCESS_TOKEN>"
